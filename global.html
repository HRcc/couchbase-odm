Example
//Model's schema property data type definition
{
    type: DataTypes.HASH_TABLE,
    schema: {
        username: {
            type: DataTypes.STRING
        },
        sex: {
            type: DataTypes.ENUM,
            enum: ['male', 'female']
        },
        friends: {
            type: DataTypes.ARRAY,
            schema: {
                type: DataTypes.COMPLEX('User')
            }
        }
    }
}
